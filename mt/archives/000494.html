#!/bin/bash

mozilla -geometry 800x800 &
PID=$!

sleep 3;

ID=`xwininfo -name "Mozilla" -int | perl -e 'while(<>) { if (/Window id: ([0-9]+)/) { print $1; } }'`

mozilla -remote openURL\(http://$1\)

sleep 3;

import -window $ID -silent $1.jpg

kill $PID
